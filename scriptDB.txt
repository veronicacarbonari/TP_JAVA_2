DROP TABLE IF EXISTS atracciones;
CREATE TABLE atracciones (
	"id_atraccion"	INTEGER NOT NULL,
	"nombre"	TEXT NOT NULL,
	"cupo"	INTEGER NOT NULL,
	"visitantes"	INTEGER,
	"costo"	INTEGER NOT NULL,
	"duracion"	INTEGER NOT NULL,
	"tipo"	INTEGER NOT NULL,
	PRIMARY KEY("id_atraccion" AUTOINCREMENT)
);

DROP TABLE IF EXISTS itinerarios;
CREATE TABLE itinerarios (
	"nombre_usuario"	TEXT NOT NULL,
	"costo_total"	INTEGER NOT NULL,
	"duracion_total"	NUMERIC NOT NULL,
	"atracciones"	INTEGER NOT NULL
	
);

DROP TABLE IF EXISTS tipo_atracciones;
CREATE TABLE tipo_atracciones (
	"id_tipo"	INTEGER NOT NULL,
	"tipo"	TEXT NOT NULL,
	PRIMARY KEY("id_tipo")
);

DROP TABLE IF EXISTS usuarios;
CREATE TABLE usuarios (
	"id_usuario"	INTEGER NOT NULL,
	"nombre"	TEXT NOT NULL,
	"tipo_preferencia"	TEXT NOT NULL,
	"tiempo"	NUMERIC NOT NULL,
	"presupuesto"	INTEGER NOT NULL,
	PRIMARY KEY("id_usuario" AUTOINCREMENT)
);
